задание 1

SELECT c.login,
	COUNT(o."inDelivery" = true)
FROM "Couriers" AS c
INNER JOIN "Orders" AS o ON c.id = o."courierId"
GROUP BY c.login;

задание 2

SELECT track,
	CASE
        WHEN finished = true THEN 2
        WHEN cancelled = true THEN -1
        WHEN "inDelivery" = true THEN 1
	ELSE 0
    	END  AS "status"
FROM "Orders";
